[docs]def mp_start():
    # starting mutiprocessing manager
    global mng
    global wrk
    mng = Popen(["python", cf._ibc + "mp/mp_manager.py"])
    mng.poll()
    time.sleep(1)  # time to create "hostname.txt" file
    # starting parallel workers
    for i in xrange(cf._nr_wrk/2):
        wrk.append(MPWorker(i))
        wrk[-1].start()
